# JS 代码规范

# 1. 引用

1.1 对所有的引用使用 const 和 let, 先定义再使用

// bad
var a = 1;
var b = 2;

// good
const a = 1;
let b = 2;

1.2 常量数据全大写

// bad
const userToken = 111111;

// good
const USER_TOKEN = 12222

# 2. 对象

2.1 对象声明

// bad
const obj = new Object()

// good
const obj = {}

2.2 禁止使用保留字做键值

// bad
const default = 'default 是 保留字'

// good
const defaults = '类似于这样'

2.3 使用对象属性值的简写(减少代码量, 描述更准确)

const  goodsName = '商品名'
// bad
const obj = {
	goodsName: goodsName
}

// good
const obj = {
    goodsName
}

# 3. 命名规则

3.1 见名知意, 需要明确知道这个变量代表的意义

// bad
const a = 'nickname'
function do() {
    // dosomething
}

// good
const nickName = 'nickName'
function setStyle() {
    // dosomething
}

# 4. 代码块

4.1 用代码块包裹

// bad
if(true) return ;

if(true) 
    return false;

// good 
if(true) {
    return false
}

# 5. 注释

5.1 TODO, !, ? 的使用

​ vscode 推荐使用 Better Comments(注释标记高亮) , TODO tree(整理项目中的 TODO)

// TODO: dosomething 记录之后要做的事情

// ! 严重, 记录严重影响逻辑但暂时无法删除的事情

// ? 标记有疑问的地方

5.2禁止多行单行注释, 多行请使用块注释

// bad
// make() returns a new element
// based on the passed in tag name
//
// @param {String} tag
// @return {Element} element
function make(tag) {

  // ...stuff...

  return element;
}

// good
/**
 * make() returns a new element
 * based on the passed in tag name
 *
 * @param {String} tag
 * @return {Element} element
 */
function make(tag) {

  // ...stuff...

  return element;
}

5.3 功能性函数的注释

/**
 *  打开 app 按钮
 *
 * @param {*} [params] 传递给 app 的参数
 * @param {*} [fn] 如不跳转 浏览器打开页  做的处理
 * @param {*} [vm] vue 实例
 */

# 7. 多状态处理

// 项目中肯定有n多状态码的switch
function switchState(state) {
    let res
    switch(state) {
    	case 1: 
        	res = '未支付'
        	break
        case 2: 
            res = '支付成功'
            break
	}
    return res
    
 	// 亦或者是这样的, 看起来就不是什么很舒服的代码.....
    if(state === 1 ) { res = '未支付'}
    else if (state === 2) {res = '支付成功'}
}
